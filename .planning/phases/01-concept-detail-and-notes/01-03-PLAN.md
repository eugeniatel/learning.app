**Create src/lib/notes.ts** — pure async functions, no "use server" directive here. Under 60 lines.

```typescript
import fs from "node:fs/promises";
import path from "node:path";
import matter from "gray-matter";
import { paths } from "./paths";

function noteFilePath(slug: string): string {
  return path.join(paths.notesDir, `${slug}.md`);
}

/** Returns the note body (without frontmatter) for a concept slug, or empty string if no note exists. */
export async function readNote(slug: string): Promise<string> {
  const file = noteFilePath(slug);
  try {
    const raw = await fs.readFile(file, "utf8");
    const { content } = matter(raw);
    return content.trim();
  } catch {
    // File does not exist yet
    return "";
  }
}

/**
 * Writes note body to notes/<slug>.md with frontmatter.
 * Creates notes/ directory if it does not exist.
 * Throws on write failure (caller handles the error).
 */
export async function saveNote(slug: string, conceptId: string, body: string): Promise<void> {
  await fs.mkdir(paths.notesDir, { recursive: true });
  const file = noteFilePath(slug);
  const updatedAt = new Date().toISOString();
  const output = matter.stringify(body, { conceptId, updatedAt });
  await fs.writeFile(file, output, "utf8");
}
```

**Create src/components/note-editor.tsx** as a Client Component. Under 200 lines.

Props:
```typescript
type Props = {
  slug: string;
  conceptId: string;
  initialBody: string;
}
```

State:
- `body: string` — current textarea value, initialized to `initialBody`
- `savedBody: string` — last successfully saved body, initialized to `initialBody`; used to compute dirty state
- `status: "idle" | "saving" | "saved" | "error"` — drive button label and error message

Dirty: `body !== savedBody`

On save click:
1. Set status = "saving"
2. Call `saveNoteAction(slug, conceptId, body)` — await it
3. On success: set `savedBody = body`, set `status = "saved"`, setTimeout 2000ms to set `status = "idle"`
4. On error: set `status = "error"`

Button label: status === "saved" ? "Saved" : "Save note"
Button variant: dirty && status !== "saved" ? "default" : "outline"
Button disabled: status === "saving"

Section label style: `text-[13px] font-medium uppercase tracking-[0.04em] text-muted-foreground mb-3`

```tsx
"use client";
import { useState } from "react";
import { Textarea } from "@/components/ui/textarea";
import { Button } from "@/components/ui/button";
import { saveNoteAction } from "@/lib/actions/save-note";

export function NoteEditor({
  slug,
  conceptId,
  initialBody,
}: {
  slug: string;
  conceptId: string;
  initialBody: string;
}) {
  const [body, setBody] = useState(initialBody);
  const [savedBody, setSavedBody] = useState(initialBody);
  const [status, setStatus] = useState<"idle" | "saving" | "saved" | "error">("idle");

  const dirty = body !== savedBody;

  async function handleSave() {
    setStatus("saving");
    try {
      await saveNoteAction(slug, conceptId, body);
      setSavedBody(body);
      setStatus("saved");
      setTimeout(() => setStatus("idle"), 2000);
    } catch {
      setStatus("error");
    }
  }

  return (
    <div>
      <p className="mb-3 text-[13px] font-medium uppercase tracking-[0.04em] text-muted-foreground">
        Notes
      </p>
      <Textarea
        value={body}
        onChange={(e) => {
          setBody(e.target.value);
          if (status === "error") setStatus("idle");
        }}
        placeholder="Write notes about this concept..."
        className="min-h-[160px] resize-y bg-card text-[15px] leading-[1.7] font-normal"
      />
      {status === "error" && (
        <p className="mt-1.5 text-[13px] text-destructive">Could not save. Try again.</p>
      )}
      <div className="mt-2 flex justify-end">
        <Button
          variant={dirty && status !== "saved" ? "default" : "outline"}
          size="sm"
          onClick={handleSave}
          disabled={status === "saving"}
          className="transition-opacity duration-[180ms] ease-out"
        >
          {status === "saved" ? "Saved" : "Save note"}
        </Button>
      </div>
    </div>
  );
}
```
